package com.jpaLab;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.LinkedList;
import java.util.Queue;

class Node {
    int idx;
    ArrayList<Node> child = new ArrayList();

    public Node(int idx) {
        this.idx = idx;
    }

    public void addChild(Node child) {
        this.child.add(child);
    }
}

class Tree {
    Node head;

    public Tree(int[] nums, int k) {
        Node[] nodes = new Node[nums.length];
        for (int i = 0; i < nums.length; i++) {
            nodes[i] = new Node(i);
        }
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == -1) {
                head = nodes[i];
                continue;
            }
            nodes[nums[i]].addChild(nodes[i]);
        }

        this.deleteChild(nodes, k);
    }

    public void deleteChild(Node[] nodes, int k) {
        for (int i = 0; i < nodes.length; i++) {
            for (int j = 0; j < nodes[i].child.size(); j++) {
                if (nodes[i].child.get(j).idx == k) {
                    nodes[i].child.remove(j);
                    break;
                }
            }
        }
    }
    public int getCount(Node node){
        int count = 0;
        Queue<Node> queue = new LinkedList<>();
        queue.add(node);

        while (!queue.isEmpty()){
            Node cur = queue.poll();
            if(cur.child.size() == 0) {
                count++;
                continue;
            }

            for (int i = 0; i < cur.child.size(); i++) {
                queue.add(cur.child.get(i));
            }
        }
        return count;
    }
}

public class Main {
    public static void main (String[] args) throws IOException {

        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int N = Integer.parseInt(br.readLine());
        String str = br.readLine();
        int K = Integer.parseInt(br.readLine());
        if (K == 0) {
            System.out.println(0);
            return;
        }

        int[] num = new int[N];
        for (int i = 0; i < str.split(" ").length; i++) {
            num[i] = Integer.parseInt(str.split(" ")[i]);
        }

        Tree tree = new Tree(num, K);
        System.out.println(tree.getCount(tree.head));
    }
}
